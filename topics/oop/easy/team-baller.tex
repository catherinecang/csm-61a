\begin{solution}[1in]
\begin{lstlisting}
class TeamBaller(Baller):
    """
    >>> Alina = BallHog('Alina')
    >>> cheerballer = TeamBaller('Alex', has_ball=True)
    >>> cheerballer.pass_ball(Alina)
    Yay!
    True
    >>> cheerballer.pass_ball(Alina)
    I don't have the ball
    False
    """
    def pass_ball(self, other):
        did_pass = Baller.pass_ball(self, other)
        if did_pass:
            print('Yay!')
        else:
            print('I don't have the ball')
        return did_pass
\end{lstlisting}
