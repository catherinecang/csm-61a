\question What Would Scheme Display?

\begin{parts}

\part \lstinline$scm> (define x 1)$
\begin{solution}[0.25in]
\lstinline$x$
\end{solution}

\part \lstinline$scm> (define p (delay (+ x 1)))$
\begin{solution}[.25in]
\lstinline$p$
\end{solution}

\part \lstinline$scm> p$
\begin{solution}[.25in]
\lstinline$#[promise (unforced)]$
\end{solution}

\part \lstinline$scm> (force p)$
\begin{solution}[.25in]
\lstinline$2$
\end{solution}

\part \lstinline$scm> (define (foo x) (+ x 10))$
\begin{solution}[0.25in]
\lstinline$foo$
\end{solution}

\part 
\begin{lstlisting}
scm> (define bar (cons-stream (foo 1) 
                              (cons-stream (foo 2) bar)))
\end{lstlisting}
\begin{solution}[0.25in]
\lstinline$bar$
\end{solution}

\part \lstinline$scm> (car bar)$
\begin{solution}[0.25in]
\lstinline$11$
\end{solution}

\part \lstinline$scm> (cdr bar)$
\begin{solution}[0.25in]
\lstinline$#[promise (unforced)]$
\end{solution}

\part \lstinline$scm> (define (foo x) (+ x 1))$
\begin{solution}[0.25in]
\lstinline$foo$
\end{solution}

\part \lstinline$scm> (cdr-stream bar)$
\begin{solution}[0.25in]
\lstinline$(3 . #[promise (unforced)])$
\end{solution}

\part \lstinline$scm> (define (foo x) (+ x 5))$
\begin{solution}[0.25in]
\lstinline$foo$
\end{solution}

\part \lstinline$scm> (car bar)$
\begin{solution}[0.25in]
\lstinline$11$
\end{solution}

\part \lstinline$scm> (cdr-stream bar)$
\begin{solution}[0.25in]
\lstinline$(3 . #[promise (unforced)])$
\end{solution}
\end{parts}
