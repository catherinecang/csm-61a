class SkipMachine:
    skip = 1
    def __init__(self, n=2):
        self.skip = n + SkipMachine.skip

    def generate(self):
        current = SkipMachine.skip
        while True:
            yield current
            current += self.skip
            SkipMachine.skip += 1
    p = SkipMachine()
    twos = p.generate()
    SkipMachine.skip += 1
    twos = p.generate()
    threes = SkipMachine(3).generate()
