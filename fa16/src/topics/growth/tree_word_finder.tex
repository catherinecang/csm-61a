\question \textbf{Orders of Growth and Trees:} Assume we are using the
non-mutable tree implementation introduced earlier. Consider the following
function:
\begin{lstlisting}
def word_finder(t, n, word):
    if root(t) == word:
        n -= 1
        if n == 0:
            return True
    for branch in branches(t):
        if word_finder(branch, n, word):
            return True
    return False
\end{lstlisting}

\begin{parts}
\part What does this function do?  \newline
\begin{solution}[0.25in]
This function take a Tree \texttt{t}, an integer \texttt{n}, and a string
\texttt{word} in as input.

Then, \texttt{word\char`_finder} returns True if any paths from the root
towards the leaves have at least \texttt{n} occurrences of the word and False
otherwise.
\end{solution}

\part If a tree has $n$ total nodes, what is the total runtime for all searches
in big-O notation?
\begin{solution}[0.25in]
$O(n)$. At worst, we must visit every node of the tree.
\end{solution}

\end{parts}
